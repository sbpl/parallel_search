258  static double computeLoSHeuristic(const StateVarsType& state_vars)
259  {
260  size_t state_key = StateKeyGenerator(state_vars);
261  if (config_.heuristic_cache.find(state_key) != config_.heuristic_cache.end())
262  {
263  return config_.heuristic_cache[state_key];
264  }
265 
266  double h = computeHeuristic(state_vars);
267  config_.heuristic_cache[state_key] = h;
268  int N = static_cast<int>(h)/9e-1;
269  Eigen::Map<const VecDf> p1(&state_vars[0], state_vars.size());
270  Eigen::Map<const VecDf> p2(&config_.goal[0], config_.goal.size());
271 
272  for (int i=0; i<N; ++i)
273  {
274  double j = i/static_cast<double>(N);
275  VecDf intp_pt = p1*(1-j) + p2*j;
276 
277  mju_copy(config_.global_d->qpos, intp_pt.data(), config_.global_m->nq);
278  mj_fwdPosition(config_.global_m, config_.global_d);
279 
280  if (config_.global_d->ncon > 0)
281  {
282  config_.heuristic_cache[state_key] = 100;
283  break;
284  }
285  }
286 
287  return config_.heuristic_cache[state_key];
288  }

321  static void setupSmplBFS()
322  {
323  Vec3f lwh;
324  lwh << config_.global_bfs_m->numeric_data[3] - config_.global_bfs_m->numeric_data[0],
325  config_.global_bfs_m->numeric_data[4] - config_.global_bfs_m->numeric_data[1],
326  config_.global_bfs_m->numeric_data[5] - config_.global_bfs_m->numeric_data[2];
327 
328  int length = static_cast<int>(lwh(0)/BFS_DISCRETIZATION)+1;
329  int width = static_cast<int>(lwh(1)/BFS_DISCRETIZATION)+1;
330  int height = static_cast<int>(lwh(2)/BFS_DISCRETIZATION)+1;
331 
332  std::vector<std::vector<int>> occupied_cells;
333  for (int i=0; i<length; ++i)
334  {
335  for (int j=0; j<width; ++j)
336  {
337  for (int k=0; k<height; ++k)
338  {
339  Vec3f xyz;
340  xyz << i*BFS_DISCRETIZATION + config_.global_bfs_m->numeric_data[0],
341  j*BFS_DISCRETIZATION + config_.global_bfs_m->numeric_data[1],
342  k*BFS_DISCRETIZATION + config_.global_bfs_m->numeric_data[2];
343 
344  VecDf fullstate(7);
345  fullstate << xyz(0), xyz(1), xyz(2), 1, 0, 0, 0;
346  mju_copy(config_.global_bfs_d->qpos, fullstate.data(), config_.global_bfs_m->nq);
347  mj_fwdPosition(config_.global_bfs_m, config_.global_bfs_d);
348 
349  if (config_.global_bfs_d->ncon > 0)
350  {
351  std::vector<int> occupied_cell;
352  occupied_cell.emplace_back(i);
353  occupied_cell.emplace_back(j);
354  occupied_cell.emplace_back(k);
355 
356  occupied_cells.push_back(occupied_cell);
357  }
358  }
359  }
360  }
361 
362  initializeBFS(length, width, height, occupied_cells);
363 
364  std::cout << "Finished setting up SMPL bfs3d environment of size " << length << "x" << width << "x" << height
365  << " cells containing " << occupied_cells.size() << " occupied cells." << std::endl;
366  }

377  static double computeBFSHeuristic(const StateVarsType& state_vars)
378  {
379  Vec3f ee_pos = getEEPosition(state_vars);
380 
381  int x = static_cast<int>((ee_pos(0) - config_.global_bfs_m->numeric_data[0]) / BFS_DISCRETIZATION);
382  int y = static_cast<int>((ee_pos(1) - config_.global_bfs_m->numeric_data[1]) / BFS_DISCRETIZATION);
383  int z = static_cast<int>((ee_pos(2) - config_.global_bfs_m->numeric_data[2]) / BFS_DISCRETIZATION);
384  double cost_per_cell = 1;
385 
386  if (!config_.bfs3d->inBounds(x, y, z)) {
387  return DINF;
388  }
389  else if (config_.bfs3d->getDistance(x, y, z) == smpl::BFS_3D::WALL) {
390  return DINF;
391  }
392  else {
393  return cost_per_cell * config_.bfs3d->getDistance(x, y, z);
394  }
395  }

429  static MatDf loadMPrims(std::string mprim_file)
430  {
431  if (!config_.global_m)
432  {
433  std::runtime_error("Attempting to load motion primitives before Mujoco model. ERROR!");
434  }
435 
437  MatDf mprims = loadEigenFromFile<MatDf>(mprim_file, ' ');
438 
440  int num_input_prim = mprims.rows();
441  mprims.conservativeResize(2*mprims.rows(), mprims.cols());
442  mprims.block(num_input_prim, 0, num_input_prim, mprims.cols()) =
443  -1*mprims.block(0, 0, num_input_prim, mprims.cols());
444 
445  return mprims;

448  static void constructActions(std::vector<std::shared_ptr<Action>>& action_ptrs,
449  ParamsType& action_params,
450  std::string& mj_modelpath, std::string& mprimpath,
451  ManipulationAction::OptVecPtrType& opt,
452  int num_threads)
453  {
455  ManipulationAction::MjModelVecType m_vec;
456  ManipulationAction::MjDataVecType d_vec;
457  for (int i=0; i<num_threads; ++i)
458  {
459  mjModel* act_m= nullptr;
460  mjData * act_d= nullptr;
461  setupMujoco(&act_m, &act_d, mj_modelpath);
462  m_vec.push_back(act_m);
463  d_vec.push_back(act_d);
464  }
465 
467  auto mprims = loadMPrims(mprimpath);
468  mprims *= (M_PI/180.0);
469  action_params["length"] = mprims.rows();
470 
471  for (int i=0; i<=action_params["length"]; ++i)
472  {
473  if (i == action_params["length"])
474  {
475  auto one_joint_action = std::make_shared<OneJointAtATime>(std::to_string(i), action_params,
476  DISCRETIZATION, mprims,
477  opt, m_vec, d_vec, num_threads, 1);
478  action_ptrs.emplace_back(one_joint_action);
479  }
480  else
481  {
482  bool is_expensive = (action_params["planner_type"] == 1) ? 1 : 0;
483  auto one_joint_action = std::make_shared<OneJointAtATime>(std::to_string(i), action_params,
484  DISCRETIZATION, mprims,
485  opt, m_vec, d_vec, num_threads, is_expensive);
486  action_ptrs.emplace_back(one_joint_action);
487  }
488  }
489 
490  // So that the adaptive primitive is tried first
491  std::reverse(action_ptrs.begin(), action_ptrs.end());
492  }

526  static void constructPlanner(std::string planner_name, std::shared_ptr<Planner>& planner_ptr, std::vector<std::shared_ptr<Action>>& action_ptrs, ParamsType& planner_params, ParamsType& action_params, BSplineOpt& opt)
527  {
528  if (planner_name == "epase")
529  planner_ptr = std::make_shared<EpasePlanner>(planner_params);
530  else if (planner_name == "gepase")
531  planner_ptr = std::make_shared<GepasePlanner>(planner_params);
532  else if (planner_name == "insat")
533  planner_ptr = std::make_shared<InsatPlanner>(planner_params);
534  else if (planner_name == "pinsat")
535  planner_ptr = std::make_shared<PinsatPlanner>(planner_params);
536  else if (planner_name == "rrt")
537  planner_ptr = std::make_shared<RrtPlanner>(planner_params);
538  else if (planner_name == "rrtconnect")
539  planner_ptr = std::make_shared<RrtConnectPlanner>(planner_params);
540  else if (planner_name == "wastar")
541  planner_ptr = std::make_shared<WastarPlanner>(planner_params);
542  else
543  throw std::runtime_error("Planner type not identified!");
544 
546  if (config_.h_mode == HeuristicMode::EUCLIDEAN)
547  {
548  planner_ptr->SetHeuristicGenerator(std::bind(computeHeuristic, std::placeholders::_1));
549  }
550  else if (config_.h_mode == HeuristicMode::LOS)
551  {
552  planner_ptr->SetHeuristicGenerator(std::bind(computeLoSHeuristic, std::placeholders::_1));
553  }
554  else if (config_.h_mode == HeuristicMode::SHIELD)
555  {
556  planner_ptr->SetHeuristicGenerator(std::bind(computeShieldHeuristic, std::placeholders::_1));
557  }
558  else if (config_.h_mode == HeuristicMode::SMPL_BFS)
559  {
560  planner_ptr->SetHeuristicGenerator(std::bind(computeBFSHeuristic, std::placeholders::_1));
561  }
562  else if (config_.h_mode == HeuristicMode::EE)
563  {
564  planner_ptr->SetHeuristicGenerator(std::bind(computeEEHeuristic, std::placeholders::_1));
565  }
566 
567 
568  planner_ptr->SetActions(action_ptrs);
569  planner_ptr->SetStateMapKeyGenerator(bind(StateKeyGenerator, std::placeholders::_1));
570  planner_ptr->SetEdgeKeyGenerator(bind(EdgeKeyGenerator, std::placeholders::_1));
571  planner_ptr->SetStateToStateHeuristicGenerator(bind(computeHeuristicStateToState, std::placeholders::_1, std::placeholders::_2));
572 
574  if (config_.goal_mode == GoalCheckerMode::CSPACE)
575  {
576  planner_ptr->SetGoalChecker(std::bind(isGoalState, std::placeholders::_1, TERMINATION_DIST));
577  }
578  else if (config_.goal_mode == GoalCheckerMode::EE)
579  {
580  planner_ptr->SetGoalChecker(std::bind(isEEGoalState, std::placeholders::_1, TERMINATION_DIST));
581  }
582 
584  if ((planner_name != "pinsat") && (planner_name != "insat"))
585  {
586  if (config_.pp_mode == PPMode::WAYPT)
587  {
588  planner_ptr->SetPostProcessor(std::bind(postProcess, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3, action_ptrs[0], opt));
589  }
590  else if (config_.pp_mode == PPMode::CONTROLPT)
591  {
592  planner_ptr->SetPostProcessor(std::bind(postProcessWithControlPoints, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3, action_ptrs[0], opt));
593  }
594  }
595  }

597  static MatDf sampleTrajectory(const drake::trajectories::BsplineTrajectory<double>& traj, double dt=1e-1)
598  {
599  MatDf sampled_traj;
600  int i=0;
