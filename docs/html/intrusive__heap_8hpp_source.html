267 template <class InputIt>
268 void intrusive_heap<T, Compare>::make_heap(InputIt first, InputIt last)
269 {
270  auto n = std::distance(first, last);
271 
272  m_data.clear();
273  m_data.reserve(n + 1);
274  m_data.push_back(nullptr);
275  m_data.insert(m_data.end(), first, last);
276 
277  for (size_type i = 1; i < m_data.size(); ++i) {
278  m_data[i]->m_heap_index = i;
279  }
280 
281  for (auto i = n >> 1; i >= 1; --i) {
282  percolate_down(i);
283  }
284 }

286 template <class T, class Compare>
287 template <class InputIt>
288 void intrusive_heap<T, Compare>::make_heap(
289  InputIt first,
290  InputIt last,
291  size_type root)
292 {
293  const auto n = std::distance(first, last);
294  printf("make heap from %d elements from %zu\n", n, root);
295  print();
296 
297  if (n <= 0) {
298  return;
299  }
300 
301  printf(" -> data[%zu] = %p\n", root, *first);
302  m_data[root] = *first;
303  m_data[root]->m_heap_index = root;
304 
305  if (n == 1) {
306  return;
307  }
308 
309  const size_type left = left_child(root);
310  const size_type right = right_child(root);
311 
312  auto f = ilog2(n) - 1;
313  size_type f2 = ipow2(f);
314  size_type l = f2 - 1 + std::min(n - 2 * f2 + 1, f2);
315  size_type r = n - 1 - l;
316 
317  InputIt new_start = std::next(first);
318  InputIt mid = std::next(new_start, l);
319 
320  make_heap(new_start, mid, left);
321  make_heap(mid, last, right);
322  percolate_down(root);
323 }

349 template <class T, class Compare>
350 inline
351 void intrusive_heap<T, Compare>::percolate_down(size_type pivot)
352 {
353  if (is_external(pivot)) {
354  return;
355  }
356 
357  size_type left = left_child(pivot);
358  size_type right = right_child(pivot);
359 
360  T* tmp = m_data[pivot];
361  while (is_internal(left)) {
362  size_type s = right;
363  if (is_external(right) || m_comp(*m_data[left], *m_data[right])) {
364  s = left;
365  }
366 
367  if (m_comp(*m_data[s], *tmp)) {
368  m_data[pivot] = m_data[s];
369  m_data[pivot]->m_heap_index = pivot;
370  pivot = s;
371  } else {
372  break;
373  }
374 
375  left = left_child(pivot);
376  right = right_child(pivot);
377  }
378  m_data[pivot] = tmp;
379  m_data[pivot]->m_heap_index = pivot;
380 }

382 template <class T, class Compare>
383 inline
384 void intrusive_heap<T, Compare>::percolate_up(size_type pivot)
385 {
386  T* tmp = m_data[pivot];
387  while (pivot != 1) {
388  size_type p = parent(pivot);
389  if (m_comp(*m_data[p], *tmp)) {
390  break;
391  }
392  m_data[pivot] = m_data[p];
393  m_data[pivot]->m_heap_index = pivot;
394  pivot = p;
395  }
396  m_data[pivot] = tmp;
397  m_data[pivot]->m_heap_index = pivot;
398 }

429 template <class T, class Compare>
430 bool intrusive_heap<T, Compare>::check_heap(size_type index) const
431 {
432  auto right = right_child(index);
433  if (is_internal(right)) {
434  // check the ordering of the parent and the right child
435  if (!m_comp(*m_data[index], *m_data[right])) {
436  return false;
437  }
438 
439  // check the tree rooted at the right child
440  if (!check_heap(right)) {
441  return false;
442  }
443  }
444 
445  auto left = left_child(index);
446  if (is_internal(left)) {
447  // check the ordering of the parent and the left child
448  if (!m_comp(*m_data[index], *m_data[left])) {
449  return false;
450  }
451 
452  // check the tree rooted at the left child
453  if (!check_heap(left)) {
454  return false;
455  }
456  }
457 
458  return true;
459 }
