119  static inline double two_pi_complement(double angle)
120  {
121  //check input conditions
122  if (angle > 2*M_PI || angle < -2.0*M_PI)
123  angle = fmod(angle, 2.0*M_PI);
124  if(angle < 0)
125  return (2*M_PI+angle);
126  else if (angle > 0)
127  return (-2*M_PI+angle);
128 
129  return(2*M_PI);
130  }

143  static bool find_min_max_delta(double from, double left_limit, double right_limit, double &result_min_delta, double &result_max_delta)
144  {
145  double delta[4];
146 
147  delta[0] = shortest_angular_distance(from,left_limit);
148  delta[1] = shortest_angular_distance(from,right_limit);
149 
150  delta[2] = two_pi_complement(delta[0]);
151  delta[3] = two_pi_complement(delta[1]);
152 
153  if(delta[0] == 0)
154  {
155  result_min_delta = delta[0];
156  result_max_delta = std::max<double>(delta[1],delta[3]);
157  return true;
158  }
159 
160  if(delta[1] == 0)
161  {
162  result_max_delta = delta[1];
163  result_min_delta = std::min<double>(delta[0],delta[2]);
164  return true;
165  }
166 
167 
168  double delta_min = delta[0];
169  double delta_min_2pi = delta[2];
170  if(delta[2] < delta_min)
171  {
172  delta_min = delta[2];
173  delta_min_2pi = delta[0];
174  }
175 
176  double delta_max = delta[1];
177  double delta_max_2pi = delta[3];
178  if(delta[3] > delta_max)
179  {
180  delta_max = delta[3];
181  delta_max_2pi = delta[1];
182  }
183 
184 
185  // printf("%f %f %f %f\n",delta_min,delta_min_2pi,delta_max,delta_max_2pi);
186  if((delta_min <= delta_max_2pi) || (delta_max >= delta_min_2pi))
187  {
188  result_min_delta = delta_max_2pi;
189  result_max_delta = delta_min_2pi;
190  if(left_limit == -M_PI && right_limit == M_PI)
191  return true;
192  else
193  return false;
194  }
195  result_min_delta = delta_min;
196  result_max_delta = delta_max;
197  return true;
198  }

224  static inline bool shortest_angular_distance_with_large_limits(double from, double to, double left_limit, double right_limit, double &shortest_angle)
225  {
226  // Shortest steps in the two directions
227  double delta = shortest_angular_distance(from, to);
228  double delta_2pi = two_pi_complement(delta);
229 
230  // "sort" distances so that delta is shorter than delta_2pi
231  if(std::fabs(delta) > std::fabs(delta_2pi))
232  std::swap(delta, delta_2pi);
233 
234  if(left_limit > right_limit) {
235  // If limits are something like [PI/2 , -PI/2] it actually means that we
236  // want rotations to be in the interval [-PI,PI/2] U [PI/2,PI], ie, the
237  // half unit circle not containing the 0. This is already gracefully
238  // handled by shortest_angular_distance_with_limits, and therefore this
239  // function should not be called at all. However, if one has limits that
240  // are larger than PI, the same rationale behind shortest_angular_distance_with_limits
241  // does not hold, ie, M_PI+x should not be directly equal to -M_PI+x.
242  // In this case, the correct way of getting the shortest solution is to
243  // properly set the limits, eg, by saying that the interval is either
244  // [PI/2, 3*PI/2] or [-3*M_PI/2, -M_PI/2]. For this reason, here we
245  // return false by default.
246  shortest_angle = delta;
247  return false;
248  }
249 
250  // Check in which direction we should turn (clockwise or counter-clockwise).
251 
252  // start by trying with the shortest angle (delta).
253  double to2 = from + delta;
254  if(left_limit <= to2 && to2 <= right_limit) {
255  // we can move in this direction: return success if the "from" angle is inside limits
256  shortest_angle = delta;
257  return left_limit <= from && from <= right_limit;
258  }
259 
260  // delta is not ok, try to move in the other direction (using its complement)
261  to2 = from + delta_2pi;
262  if(left_limit <= to2 && to2 <= right_limit) {
263  // we can move in this direction: return success if the "from" angle is inside limits
264  shortest_angle = delta_2pi;
265  return left_limit <= from && from <= right_limit;
266  }
267 
268  // nothing works: we always go outside limits
269  shortest_angle = delta; // at least give some "coherent" result
270  return false;
271  }

291  static inline bool shortest_angular_distance_with_limits(double from, double to, double left_limit, double right_limit, double &shortest_angle)
292  {
293 
294  double min_delta = -2*M_PI;
295  double max_delta = 2*M_PI;
296  double min_delta_to = -2*M_PI;
297  double max_delta_to = 2*M_PI;
298  bool flag = find_min_max_delta(from,left_limit,right_limit,min_delta,max_delta);
299  double delta = shortest_angular_distance(from,to);
300  double delta_mod_2pi = two_pi_complement(delta);
301 
302 
303  if(flag)//from position is within the limits
304  {
305  if(delta >= min_delta && delta <= max_delta)
306  {
307  shortest_angle = delta;
308  return true;
309  }
310  else if(delta_mod_2pi >= min_delta && delta_mod_2pi <= max_delta)
311  {
312  shortest_angle = delta_mod_2pi;
313  return true;
314  }
315  else //to position is outside the limits
316  {
317  find_min_max_delta(to,left_limit,right_limit,min_delta_to,max_delta_to);
318  if(fabs(min_delta_to) < fabs(max_delta_to))
319  shortest_angle = std::max<double>(delta,delta_mod_2pi);
320  else if(fabs(min_delta_to) > fabs(max_delta_to))
321  shortest_angle = std::min<double>(delta,delta_mod_2pi);
322  else
323  {
324  if (fabs(delta) < fabs(delta_mod_2pi))
325  shortest_angle = delta;
326  else
327  shortest_angle = delta_mod_2pi;
328  }
329  return false;
330  }
331  }
332  else // from position is outside the limits
333  {
334  find_min_max_delta(to,left_limit,right_limit,min_delta_to,max_delta_to);
335 
336  if(fabs(min_delta) < fabs(max_delta))
337  shortest_angle = std::min<double>(delta,delta_mod_2pi);
338  else if (fabs(min_delta) > fabs(max_delta))
339  shortest_angle = std::max<double>(delta,delta_mod_2pi);
340  else
341  {
342  if (fabs(delta) < fabs(delta_mod_2pi))
343  shortest_angle = delta;
344  else
345  shortest_angle = delta_mod_2pi;
346  }
347  return false;
348  }
349 
350  shortest_angle = delta;
351  return false;
352  }

356  static inline double interpolateAngle(double from, double to, double amount, bool in_degrees=false)
357  {
358  // convert to radians if not
359  double from_rad = in_degrees? from_degrees(from): from;
360  double to_rad = in_degrees? from_degrees(to): to;
361 
362  // wrap tp [-pi/2, pi/2]
363  from_rad = normalize_angle(from_rad);
364  to_rad = normalize_angle(to_rad);
365 
366  // find the shortest angle distance from from_rad to to_rad
367  double diff_rad = shortest_angular_distance(from_rad, to_rad);
368 
369  // the interpolated angle in [-pi/2, pi/2]
370  double result = normalize_angle(from_rad + amount*diff_rad);
371 
372  // return in radians or degrees based on user preference
373  return (in_degrees)? to_degrees(result): result;
374  }

402  static inline void angLinearize(VecDf& q1, VecDf& q2)
403  {
404  double diff;
405  for (int i=0; i<q1.size(); ++i)
406  {
407  // bring everything to [-pi/2, pi/2] range
408  q1(i) = normalize_angle(q1(i));
409  q2(i) = normalize_angle(q2(i));
410  // direct diff
411  double diff = fabs(q2(i)-q1(i));
412  // shortest angle diff
413  double s = fabs(shortest_angular_distance(q1(i), q2(i)));
414  // only if the direct diff and shortest ang diff are different (crossing pi in [-pi/2, pi/2] range)
415  if (fabs(diff-s) > 0.001)
416  {
417  q1(i) = normalize_angle_positive(q1(i));
418  q2(i) = normalize_angle_positive(q2(i));
419  }
420  }
421  }
